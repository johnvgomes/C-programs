#include <stdio.h>
#include <locale.h>

main ()
{
	
	setlocale(LC_ALL,"portuguese");
	
	int n[12];
	int x[12];
	int y[12];
	int i;
	int j;
	
	for (i=0;i<12;i++)
	{
		x[i]=0;
		y[i]=0;
	}
	
	for (i=0;i<12;i++)
	{
		
		printf ("Digite um número para inserir na posição %i do vetor N\n",(i+1));
		scanf ("%i", &n[i]);
		
		if ((n[i] % 3 == 0) && (n[i] % 2 == 0))
		{
			
			x[i] = n[i];
			
		}
		
		if ((n[i] % 5 == 0) && (n[i] % 2 != 0))
		{
			
			y[i] = n[i];
			
		}
		
	}
	
	printf ("O vetor X é:\n");
	for (i=0;i<12;i++)
	{
		
		printf ("%i ",x[i]);
		
	}

	{
		
		printf ("%i ",y[i]);
		
	}
