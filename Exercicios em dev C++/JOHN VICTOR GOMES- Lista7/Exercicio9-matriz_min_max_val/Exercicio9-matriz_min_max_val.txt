#include <stdio.h>
#include <locale.h>
//#include <stdlib.h>

main ()
{
	
	setlocale(LC_ALL,"portuguese");
	
	int matriz[7][4];
	int i;
	int j;
	int menor=99999;
	int posicaoi=0;
	int posicaoj=0;
	
	for (i=0;i<7;i++)
	{
		
		for (j=0;j<4;j++)
		{
			
			printf ("Digite um valor para a posição (%i,%i)\n",(i+1),(j+1));
			scanf ("%i", &matriz[i][j]);
			
			//matriz[i][j]=rand()%20;
			
			if(matriz[i][j] < menor)
			{
			
				menor = matriz[i][j];
				posicaoi = i;
				posicaoj = j;
				
			}
			
			
		}
		
	}
	
	printf ("Matriz Original\n");
	
	for (i=0;i<7;i++)
	{
		
		for (j=0;j<4;j++)
		{
			
			printf ("%i\t",matriz[i][j]);
			
		}	
		
		printf ("\n");
		
	}
	
	printf ("O menor número digitado é %i\n",menor);
	printf ("Posição: %i,%i\n",(posicaoi+1),(posicaoj+1));
	
	
}
